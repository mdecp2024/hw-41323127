import matplotlib.pyplot as plt
import numpy as np

# 設定學號最後四碼為 3127，即左上方座標 (31, 27)
x_offset = 31
y_offset = 27

# 迷宮的尺寸與設置
maze_width = 20
maze_height = 20

# 創建一個畫布，背景為白色
fig, ax = plt.subplots(figsize=(10, 10))

# 定義迷宮的牆壁 (黑色像素) 和路徑
maze = np.zeros((maze_height, maze_width))

# 設定一些牆壁 (假設這是一個簡單的迷宮設計)
maze[1:18, 5] = 1
maze[1:18, 10] = 1
maze[1:4, 1:10] = 1
maze[12:18, 10:19] = 1

# 畫出迷宮的牆壁 (黑色像素)
for y in range(maze_height):
    for x in range(maze_width):
        if maze[y, x] == 1:
            ax.add_patch(plt.Rectangle((x, maze_height - y - 1), 1, 1, color='black'))

# 標示學號最後四碼作為左上角座標 (31, 27)
ax.text(x_offset - 1, maze_height - y_offset, f'({x_offset}, {y_offset})', color='red', fontsize=12)

# 模擬一條從入口到出口的路徑 (紅色連續線段)
path_x = [2, 3, 3, 4, 5, 6, 6, 7, 8, 9, 9, 10, 11, 11, 12, 13, 14, 15, 16, 17]
path_y = [0, 1, 2, 3, 3, 4, 5, 5, 6, 6, 7, 8, 8, 9, 10, 11, 12, 13, 14, 15]
ax.plot(path_x, [maze_height - y - 1 for y in path_y], color='red', lw=2)

# 設定畫布範圍與座標軸
ax.set_xlim(0, maze_width)
ax.set_ylim(0, maze_height)
ax.set_aspect('equal')

# 去除座標軸
ax.axis('off')

# 顯示結果
plt.show()
