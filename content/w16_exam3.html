from browser import document, html
import math
  
# 初始化 HTML，確保 canvas 元素存在
def setup_canvas():
    if not document.querySelector("#my_canvas"):
        canvas = html.CANVAS(id="my_canvas", width=550, height=550)
        document <= canvas
  
setup_canvas()
 
# Set up canvas
canvas = document["my_canvas"]
ctx = canvas.getContext("2d")  #由2D來進行繪圖
 
# Set origin (31, 27)
origin_x, origin_y = 31, 27  #設定原點座標為(31,27)
 
# Label origin(標記原點)  
ctx.font = "12px Arial"  #字體大小:12  字型:Arial
ctx.fillStyle = "red"  # 字體顏色改為紅色
ctx.fillText("(31, 27)", origin_x + 5, origin_y - 15)  # ctx.fillText用以描繪文字

# Draw first set of lines
ctx.beginPath()  #啟用一個新的路徑
ctx.moveTo(origin_x, origin_y)  
ctx.lineTo(200, origin_y)  
ctx.moveTo(300, origin_y)  
ctx.lineTo(500, origin_y)  
ctx.moveTo(150, 100)  
ctx.lineTo(400, 100) 
ctx.moveTo(150, 300)  
ctx.lineTo(400, 300) 
ctx.moveTo(origin_x, 500)
ctx.lineTo(200, 500)  
ctx.moveTo(300, 500)  
ctx.lineTo(500, 500) 
ctx.moveTo(200, 400)  
ctx.lineTo(300, 400)
ctx.moveTo(origin_x, 200)  
ctx.lineTo(200, 200)
ctx.strokeStyle = "black"
ctx.lineWidth = 2
ctx.stroke() 

# Draw second set of lines
ctx.beginPath()  #啟用一個新的路徑
ctx.moveTo(origin_x, origin_y)  
ctx.lineTo(origin_x, 500)  
ctx.moveTo(500, origin_y)  
ctx.lineTo(500, 500) 
ctx.moveTo(200, 100)  
ctx.lineTo(200, 200) 
ctx.moveTo(400, 100)  
ctx.lineTo(400, 200) 
ctx.moveTo(300, 200)  
ctx.lineTo(300, 300) 
ctx.moveTo(400, 300)  
ctx.lineTo(400, 500) 
ctx.moveTo(200, 400)  
ctx.lineTo(200, 500) 
ctx.moveTo(150, 300)  
ctx.lineTo(150, 400)
ctx.strokeStyle = "black"
ctx.lineWidth = 2
ctx.stroke() 

# Draw third set of lines (last path)
ctx.beginPath()  #啟用一個新的路徑
ctx.moveTo(250, origin_y)  
ctx.lineTo(250, 50)  
ctx.moveTo(250, 50)  
ctx.lineTo(450, 50)  
ctx.moveTo(450, 50)  
ctx.lineTo(450, 250)
ctx.moveTo(450, 250)  
ctx.lineTo(350, 250)
ctx.moveTo(350, 250)  
ctx.lineTo(350, 150)
ctx.moveTo(350, 150)  
ctx.lineTo(250, 150)
ctx.moveTo(250, 150)  
ctx.lineTo(250, 250)
ctx.moveTo(250, 250)  
ctx.lineTo(50, 250)
ctx.moveTo(50, 250)  
ctx.lineTo(50, 450)
ctx.moveTo(50, 450)  
ctx.lineTo(175, 450)
ctx.moveTo(175, 450)  
ctx.lineTo(175, 350)
ctx.moveTo(175, 350)  
ctx.lineTo(350, 350)
ctx.moveTo(350, 350)  
ctx.lineTo(350, 450)
ctx.moveTo(350, 450)  
ctx.lineTo(250, 450)
ctx.moveTo(250, 450)  
ctx.lineTo(250, 500)
ctx.strokeStyle = "red"  # Set stroke color to red
ctx.lineWidth = 2
ctx.stroke()  # Draw the last path in red
